import flux.job
from pyiron_atomistics import Project

pr = Project('flux')
pr.remove_jobs(recursive=True, silently=True)

structure = pr.create.structure.ase.bulk("Al", cubic=True)

Use flux as queuing system#

job = pr.create.job.Lammps("lmp_with_queue")
job.executable.version = "2020.03.03_flux"
job.structure = structure
job.server.queue = "flux"
job.server.cores = 2
job.run()

with flux.job.FluxExecutor() as exe:

    job.server.executor = exe
    job.server.cores = 2
    job.run()
